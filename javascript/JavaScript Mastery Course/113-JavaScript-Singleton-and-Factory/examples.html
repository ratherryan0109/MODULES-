Example 5: Lazy initialization

Lazy initialization

Demonstrating lazy initialization using lazy initialization.

// Real-world scenario using lazy initialization
class JavaScriptSingletonandFactoryHandler {
  constructor() {
    this.cache = new Map();
  }
  add(key, value) {
    this.cache.set(key, value);
    console.log(`Added: ${key} => ${value}`);
  }
  get(key) {
    return this.cache.get(key);
  }
  getAll() {
    return Array.from(this.cache.entries());
  }
}

const ex5_handler = new JavaScriptSingletonandFactoryHandler();

Example 7: Error Handling

Factory method

Demonstrating error handling using factory method.

// Error handling patterns for javascript singleton and factory
function safeExecute113(fn, fallback) {
  try {
    const result = fn();
    return { success: true, data: result };
  } catch (error) {
    console.error("Error:", error.message);
    return { success: false, data: fallback };
  }
}

// Test with valid operation
const ex7_result1 = safeExecute113(() => {
  return "javascript singleton and factory executed successfully";
}, "fallback");

// Test with failing operation
const ex7_result2 = safeExecute113(() => {
  throw new Error("Test error");
}, "fallback value");

console.log("Result 1:", ex7_result1);
console.log("Result 2:", ex7_result2);
document.getElementById("ex7_output").textContent =
  "Success: " + ex7_result1.data + " | Fallback: " + ex7_result2.data;
