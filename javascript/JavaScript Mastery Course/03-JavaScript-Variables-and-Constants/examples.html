Example 5: Variable hoisting

Variable hoisting

Demonstrating variable hoisting using variable hoisting.

// Real-world scenario using variable hoisting
class JavaScriptVariablesandConstantsHandler {
  constructor() {
    this.cache = new Map();
  }
  add(key, value) {
    this.cache.set(key, value);
    console.log(`Added: ${key} => ${value}`);
  }
  get(key) {
    return this.cache.get(key);
  }
  getAll() {
    return Array.from(this.cache.entries());
  }
}

const ex5_handler = new JavaScriptVariablesandConstantsHandler();

Example 7: Error Handling

Naming conventions (camelCase)

Demonstrating error handling using naming conventions (camelcase).

// Error handling patterns for javascript variables and constants
function safeExecute03(fn, fallback) {
  try {
    const result = fn();
    return { success: true, data: result };
  } catch (error) {
    console.error("Error:", error.message);
    return { success: false, data: fallback };
  }
}

// Test with valid operation
const ex7_result1 = safeExecute03(() => {
  return "javascript variables and constants executed successfully";
}, "fallback");

// Test with failing operation
const ex7_result2 = safeExecute03(() => {
  throw new Error("Test error");
}, "fallback value");

console.log("Result 1:", ex7_result1);
console.log("Result 2:", ex7_result2);
document.getElementById("ex7_output").textContent =
  "Success: " + ex7_result1.data + " | Fallback: " + ex7_result2.data;
