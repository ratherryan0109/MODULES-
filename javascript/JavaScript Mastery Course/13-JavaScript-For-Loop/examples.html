Example 5: Nested for loops

Nested for loops

Demonstrating nested for loops using nested for loops.

// Real-world scenario using nested for loops
class JavaScriptForLoopHandler {
  constructor() {
    this.cache = new Map();
  }
  add(key, value) {
    this.cache.set(key, value);
    console.log(`Added: ${key} => ${value}`);
  }
  get(key) {
    return this.cache.get(key);
  }
  getAll() {
    return Array.from(this.cache.entries());
  }
}

const ex5_handler = new JavaScriptForLoopHandler();

Demonstrating error handling using break and continue in for loops.

// Error handling patterns for javascript for loop
function safeExecute13(fn, fallback) {
  try {
    const result = fn();
    return { success: true, data: result };
  } catch (error) {
    console.error("Error:", error.message);
    return { success: false, data: fallback };
  }
}

// Test with valid operation
const ex7_result1 = safeExecute13(() => {
  return "javascript for loop executed successfully";
}, "fallback");

// Test with failing operation
const ex7_result2 = safeExecute13(() => {
  throw new Error("Test error");
}, "fallback value");
