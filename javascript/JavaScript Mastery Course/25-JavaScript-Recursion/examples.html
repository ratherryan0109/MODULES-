Example 5: Call stack in recursion

Call stack in recursion

Demonstrating call stack in recursion using call stack in recursion.

// Real-world scenario using call stack in recursion
class JavaScriptRecursionHandler {
  constructor() {
    this.cache = new Map();
  }
  add(key, value) {
    this.cache.set(key, value);
    console.log(`Added: ${key} => ${value}`);
  }
  get(key) {
    return this.cache.get(key);
  }
  getAll() {
    return Array.from(this.cache.entries());
  }
}

Example 7: Error Handling

Tail recursion

Demonstrating error handling using tail recursion.

// Error handling patterns for javascript recursion
function safeExecute25(fn, fallback) {
  try {
    const result = fn();
    return { success: true, data: result };
  } catch (error) {
    console.error("Error:", error.message);
    return { success: false, data: fallback };
  }
}

// Test with valid operation
const ex7_result1 = safeExecute25(() => {
  return "javascript recursion executed successfully";
}, "fallback");

// Test with failing operation
const ex7_result2 = safeExecute25(() => {
  throw new Error("Test error");
}, "fallback value");

console.log("Result 1:", ex7_result1);
console.log("Result 2:", ex7_result2);
document.getElementById("ex7_output").textContent =
  "Success: " + ex7_result1.data + " | Fallback: " + ex7_result2.data;
