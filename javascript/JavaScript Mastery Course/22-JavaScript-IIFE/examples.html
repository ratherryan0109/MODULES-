Example 5: Private variables and methods

Private variables and methods

Demonstrating private variables and methods using private variables and methods.

// Real-world scenario using private variables and methods
class JavaScriptIIFEHandler {
  constructor() {
    this.cache = new Map();
  }
  add(key, value) {
    this.cache.set(key, value);
    console.log(`Added: ${key} => ${value}`);
  }
  get(key) {
    return this.cache.get(key);
  }
  getAll() {
    return Array.from(this.cache.entries());
  }
}

const ex5_handler = new JavaScriptIIFEHandler();

ex5_handler.add("concept2", "Module pattern with IIFE");

Example 7: Error Handling

IIFE with arrow functions

Demonstrating error handling using iife with arrow functions.

// Error handling patterns for javascript iife
function safeExecute22(fn, fallback) {
  try {
    const result = fn();
    return { success: true, data: result };
  } catch (error) {
    console.error("Error:", error.message);
    return { success: false, data: fallback };
  }
}

// Test with valid operation
const ex7_result1 = safeExecute22(() => {
  return "javascript iife executed successfully";
}, "fallback");

// Test with failing operation
const ex7_result2 = safeExecute22(() => {
  throw new Error("Test error");
}, "fallback value");

console.log("Result 1:", ex7_result1);
console.log("Result 2:", ex7_result2);
document.getElementById("ex7_output").textContent =
  "Success: " + ex7_result1.data + " | Fallback: " + ex7_result2.data;
Click "Run Example" to see output
