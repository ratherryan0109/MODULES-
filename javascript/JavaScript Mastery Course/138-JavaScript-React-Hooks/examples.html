Example 5: useContext for global state

useContext for global state

Demonstrating usecontext for global state using usecontext for global state.

// Real-world scenario using usecontext for global state
class JavaScriptReactHooksHandler {
  constructor() {
    this.cache = new Map();
  }
  add(key, value) {
    this.cache.set(key, value);
    console.log(`Added: ${key} => ${value}`);
  }
  get(key) {
    return this.cache.get(key);
  }
  getAll() {
    return Array.from(this.cache.entries());
  }
}

const ex5_handler = new JavaScriptReactHooksHandler();

ex5_handler.add("concept2", "useEffect lifecycle");

Example 7: Error Handling

useMemo for memoization

Demonstrating error handling using usememo for memoization.

// Error handling patterns for javascript react hooks
function safeExecute138(fn, fallback) {
  try {
    const result = fn();
    return { success: true, data: result };
  } catch (error) {
    console.error("Error:", error.message);
    return { success: false, data: fallback };
  }
}

// Test with valid operation
const ex7_result1 = safeExecute138(() => {
  return "javascript react hooks executed successfully";
}, "fallback");

// Test with failing operation
const ex7_result2 = safeExecute138(() => {
  throw new Error("Test error");
}, "fallback value");

console.log("Result 1:", ex7_result1);
console.log("Result 2:", ex7_result2);
document.getElementById("ex7_output").textContent =
  "Success: " + ex7_result1.data + " | Fallback: " + ex7_result2.data;
