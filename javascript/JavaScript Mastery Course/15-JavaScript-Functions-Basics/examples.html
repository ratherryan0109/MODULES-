Example 5: Parameters and arguments

Parameters and arguments

Demonstrating parameters and arguments using parameters and arguments.

// Real-world scenario using parameters and arguments
class JavaScriptFunctionsBasicsHandler {
  constructor() {
    this.cache = new Map();
  }
  add(key, value) {
    this.cache.set(key, value);
    console.log(`Added: ${key} => ${value}`);
  }
  get(key) {
    return this.cache.get(key);
  }
  getAll() {
    return Array.from(this.cache.entries());
  }
}

const ex5_handler = new JavaScriptFunctionsBasicsHandler();

Example 7: Error Handling

Default parameters

Demonstrating error handling using default parameters.

// Error handling patterns for javascript functions basics
function safeExecute15(fn, fallback) {
  try {
    const result = fn();
    return { success: true, data: result };
  } catch (error) {
    console.error("Error:", error.message);
    return { success: false, data: fallback };
  }
}

// Test with valid operation
const ex7_result1 = safeExecute15(() => {
  return "javascript functions basics executed successfully";
}, "fallback");

// Test with failing operation
const ex7_result2 = safeExecute15(() => {
  throw new Error("Test error");
}, "fallback value");

console.log("Result 1:", ex7_result1);
console.log("Result 2:", ex7_result2);
document.getElementById("ex7_output").textContent =
  "Success: " + ex7_result1.data + " | Fallback: " + ex7_result2.data;
Click "Run Example" to see output
