Example 5: Drag and drop file handling

Drag and drop file handling

Demonstrating drag and drop file handling using drag and drop file handling.

// Real-world scenario using drag and drop file handling
class JavaScriptFileAPIHandler {
  constructor() {
    this.cache = new Map();
  }
  add(key, value) {
    this.cache.set(key, value);
    console.log(`Added: ${key} => ${value}`);
  }
  get(key) {
    return this.cache.get(key);
  }
  getAll() {
    return Array.from(this.cache.entries());
  }
}

const ex5_handler = new JavaScriptFileAPIHandler();
ex5_handler.add("concept1", "File object and FileList");
ex5_handler.add("concept2", "FileReader methods (readAsText, readAsDataURL)");

Example 7: Error Handling

File upload with Fetch

Demonstrating error handling using file upload with fetch.

// Error handling patterns for javascript file api
function safeExecute126(fn, fallback) {
  try {
    const result = fn();
    return { success: true, data: result };
  } catch (error) {
    console.error("Error:", error.message);
    return { success: false, data: fallback };
  }
}

// Test with valid operation
const ex7_result1 = safeExecute126(() => {
  return "javascript file api executed successfully";
}, "fallback");

// Test with failing operation
const ex7_result2 = safeExecute126(() => {
  throw new Error("Test error");
}, "fallback value");

console.log("Result 1:", ex7_result1);
console.log("Result 2:", ex7_result2);
document.getElementById("ex7_output").textContent =
  "Success: " + ex7_result1.data + " | Fallback: " + ex7_result2.data;
