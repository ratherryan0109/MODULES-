Example 5: Currying with bind()

Currying with bind()

Demonstrating currying with bind() using currying with bind().

// Real-world scenario using currying with bind()
class JavaScriptCurryingHandler {
  constructor() {
    this.cache = new Map();
  }
  add(key, value) {
    this.cache.set(key, value);
    console.log(`Added: ${key} => ${value}`);
  }
  get(key) {
    return this.cache.get(key);
  }
  getAll() {
    return Array.from(this.cache.entries());
  }
}

const ex5_handler = new JavaScriptCurryingHandler();

Example 7: Error Handling

Partial application vs currying

Demonstrating error handling using partial application vs currying.

// Error handling patterns for javascript currying
function safeExecute82(fn, fallback) {
  try {
    const result = fn();
    return { success: true, data: result };
  } catch (error) {
    console.error("Error:", error.message);
    return { success: false, data: fallback };
  }
}

// Test with valid operation
const ex7_result1 = safeExecute82(() => {
  return "javascript currying executed successfully";
}, "fallback");

// Test with failing operation
const ex7_result2 = safeExecute82(() => {
  throw new Error("Test error");
}, "fallback value");

console.log("Result 1:", ex7_result1);
console.log("Result 2:", ex7_result2);
document.getElementById("ex7_output").textContent =
  "Success: " + ex7_result1.data + " | Fallback: " + ex7_result2.data;
