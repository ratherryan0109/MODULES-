// Real-world scenario using color picker and brush size
class JavaScriptWhiteboardAppHandler {
  constructor() {
    this.cache = new Map();
  }
  add(key, value) {
    this.cache.set(key, value);
    console.log(`Added: ${key} => ${value}`);
  }
  get(key) {
    return this.cache.get(key);
  }
  getAll() {
    return Array.from(this.cache.entries());
  }
}

const ex5_handler = new JavaScriptWhiteboardAppHandler();
ex5_handler.add("concept1", "Canvas drawing tools");
ex5_handler.add("concept2", "Shape tools (rectangle, circle, line)");

Example 7: Error Handling

Undo/redo system

Demonstrating error handling using undo/redo system.

// Error handling patterns for javascript whiteboard app
function safeExecute153(fn, fallback) {
  try {
    const result = fn();
    return { success: true, data: result };
  } catch (error) {
    console.error("Error:", error.message);
    return { success: false, data: fallback };
  }
}

// Test with valid operation
const ex7_result1 = safeExecute153(() => {
  return "javascript whiteboard app executed successfully";
}, "fallback");

// Test with failing operation
const ex7_result2 = safeExecute153(() => {
  throw new Error("Test error");
}, "fallback value");

console.log("Result 1:", ex7_result1);
console.log("Result 2:", ex7_result2);
document.getElementById("ex7_output").textContent =
  "Success: " + ex7_result1.data + " | Fallback: " + ex7_result2.data;
Click "Run Example" to see output
