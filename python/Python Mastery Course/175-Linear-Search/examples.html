def demonstrate_linear_search_algorithm():
    """Demonstrate Linear search algorithm with Python code."""
    # Initialize sample data
    data: list = [1, 2, 3, 4, 5]
    print(f"Original data: {data}")

    # Apply linear search algorithm pattern
    result: list = process_data(data)
