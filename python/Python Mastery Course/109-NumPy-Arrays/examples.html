    """Demonstrate np.array() from Python lists with Python code."""
    # Initialize sample data
    data: list = [1, 2, 3, 4, 5]
    print(f"Original data: {data}")

    # Apply np.array() from python lists pattern
    result: list = process_data(data)
    print(f"Processed result: {result}")

    return result
