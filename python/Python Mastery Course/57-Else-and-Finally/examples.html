Example 1: finally block for cleanup Basic

Core foundation for understanding this topic. This example demonstrates how to work with finally block for cleanup in Python.


# Example 1: finally block for cleanup
# Core foundation for understanding this topic

def demonstrate_finally_block_for_cleanup():
    """Demonstrate finally block for cleanup with Python code."""
    # Initialize sample data
    data: list = [1, 2, 3, 4, 5]
    print(f"Original data: {data}")

    # Apply finally block for cleanup pattern
    result: list = process_data(data)
    print(f"Processed result: {result}")

    return result

def process_data(items: list) -> list:
    """Process items according to finally block for cleanup rules."""
    # Processing logic
    return [item * 2 for item in items]

if __name__ == "__main__":
    demonstrate_finally_block_for_cleanup()
