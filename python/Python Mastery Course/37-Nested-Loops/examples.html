Example 1: Nested for loops (matrices) Basic

Core foundation for understanding this topic. This example demonstrates how to work with nested for loops (matrices) in Python.


# Example 1: Nested for loops (matrices)
# Core foundation for understanding this topic

def demonstrate_nested_for_loops_matrices():
    """Demonstrate Nested for loops (matrices) with Python code."""
    # Initialize sample data
    data: list = [1, 2, 3, 4, 5]
    print(f"Original data: {data}")

    # Apply nested for loops (matrices) pattern
    result: list = process_data(data)
    print(f"Processed result: {result}")

    return result

def process_data(items: list) -> list:
    """Process items according to nested for loops (matrices) rules."""
    # Processing logic
    return [item * 2 for item in items]

if __name__ == "__main__":
    demonstrate_nested_for_loops_matrices()
$ python example.py
Original data: [1, 2, 3, 4, 5]
Processed result: [2, 4, 6, 8, 10]
