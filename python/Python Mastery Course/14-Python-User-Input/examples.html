Example 1: input() function Basic

Core foundation for understanding this topic. This example demonstrates how to work with input() function in Python.


# Example 1: input() function
# Core foundation for understanding this topic

def demonstrate_input_function():
    """Demonstrate input() function with Python code."""
    # Initialize sample data
    data: list = [1, 2, 3, 4, 5]
    print(f"Original data: {data}")

    # Apply input() function pattern
    result: list = process_data(data)
    print(f"Processed result: {result}")

    return result

def process_data(items: list) -> list:
    """Process items according to input() function rules."""
    # Processing logic
    return [item * 2 for item in items]

if __name__ == "__main__":
    demonstrate_input_function()
$ python example.py
Original data: [1, 2, 3, 4, 5]
Processed result: [2, 4, 6, 8, 10]
