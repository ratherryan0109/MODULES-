# Example 1: Binary modes ("rb", "wb", "ab")
# Core foundation for understanding this topic

def demonstrate_binary_modes_rb_wb_ab():
    """Demonstrate Binary modes ("rb", "wb", "ab") with Python code."""
    # Initialize sample data
    data: list = [1, 2, 3, 4, 5]
    print(f"Original data: {data}")

    # Apply binary modes ("rb", "wb", "ab") pattern
    result: list = process_data(data)
    print(f"Processed result: {result}")

    return result

def process_data(items: list) -> list:
    """Process items according to binary modes ("rb", "wb", "ab") rules."""
    # Processing logic
