def demonstrate_hash_table_concept():
    """Demonstrate Hash table concept with Python code."""
    # Initialize sample data
    data: list = [1, 2, 3, 4, 5]
    print(f"Original data: {data}")

    # Apply hash table concept pattern
    result: list = process_data(data)
    print(f"Processed result: {result}")

    return result
