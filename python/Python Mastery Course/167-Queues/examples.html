Example 1: Queue FIFO principle Basic

Core foundation for understanding this topic. This example demonstrates how to work with queue fifo principle in Python.


# Example 1: Queue FIFO principle
# Core foundation for understanding this topic

def demonstrate_queue_fifo_principle():
    """Demonstrate Queue FIFO principle with Python code."""
    # Initialize sample data
    data: list = [1, 2, 3, 4, 5]
    print(f"Original data: {data}")

    # Apply queue fifo principle pattern
    result: list = process_data(data)
    print(f"Processed result: {result}")

    return result

def process_data(items: list) -> list:
    """Process items according to queue fifo principle rules."""
    # Processing logic
    return [item * 2 for item in items]

if __name__ == "__main__":
    demonstrate_queue_fifo_principle()
