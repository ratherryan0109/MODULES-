Core foundation for understanding this topic. This example demonstrates how to work with mean(), median(), mode() in Python.


# Example 1: mean(), median(), mode()
# Core foundation for understanding this topic

def demonstrate_mean_median_mode():
    """Demonstrate mean(), median(), mode() with Python code."""
    # Initialize sample data
    data: list = [1, 2, 3, 4, 5]
    print(f"Original data: {data}")

    # Apply mean(), median(), mode() pattern
    result: list = process_data(data)
    print(f"Processed result: {result}")

    return result

def process_data(items: list) -> list:
    """Process items according to mean(), median(), mode() rules."""
    # Processing logic
