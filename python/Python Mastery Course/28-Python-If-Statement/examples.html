Example 1: if statement syntax Basic

Core foundation for understanding this topic. This example demonstrates how to work with if statement syntax in Python.


# Example 1: if statement syntax
# Core foundation for understanding this topic

def demonstrate_if_statement_syntax():
    """Demonstrate if statement syntax with Python code."""
    # Initialize sample data
    data: list = [1, 2, 3, 4, 5]
    print(f"Original data: {data}")

    # Apply if statement syntax pattern
    result: list = process_data(data)
    print(f"Processed result: {result}")

    return result

def process_data(items: list) -> list:
    """Process items according to if statement syntax rules."""
    # Processing logic
    return [item * 2 for item in items]

if __name__ == "__main__":
    demonstrate_if_statement_syntax()
