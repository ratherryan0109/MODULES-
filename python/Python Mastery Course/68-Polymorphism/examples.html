Example 1: Duck typing philosophy Basic

Core foundation for understanding this topic. This example demonstrates how to work with duck typing philosophy in Python.


# Example 1: Duck typing philosophy
# Core foundation for understanding this topic

def demonstrate_duck_typing_philosophy():
    """Demonstrate Duck typing philosophy with Python code."""
    # Initialize sample data
    data: list = [1, 2, 3, 4, 5]
    print(f"Original data: {data}")

    # Apply duck typing philosophy pattern
    result: list = process_data(data)
    print(f"Processed result: {result}")

    return result

def process_data(items: list) -> list:
    """Process items according to duck typing philosophy rules."""
    # Processing logic
    return [item * 2 for item in items]

if __name__ == "__main__":
    demonstrate_duck_typing_philosophy()
