# Core foundation for understanding this topic

def demonstrate_variance_calculation():
    """Demonstrate Variance (σ²) calculation with Python code."""
    # Initialize sample data
    data: list = [1, 2, 3, 4, 5]
    print(f"Original data: {data}")

    # Apply variance (σ²) calculation pattern
    result: list = process_data(data)
    print(f"Processed result: {result}")

    return result
