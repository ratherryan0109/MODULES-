# Example 1: Multiple plt.plot() calls

def demonstrate_multiple_plt_plot_calls():
    """Demonstrate Multiple plt.plot() calls with Python code."""
    # Initialize sample data
    data: list = [1, 2, 3, 4, 5]
    print(f"Original data: {data}")

    # Apply multiple plt.plot() calls pattern
    result: list = process_data(data)
    print(f"Processed result: {result}")

    return result
