# Example 1: Node class definition
# Core foundation for understanding this topic

def demonstrate_node_class_definition():
    """Demonstrate Node class definition with Python code."""
    # Initialize sample data
    data: list = [1, 2, 3, 4, 5]
    print(f"Original data: {data}")

    # Apply node class definition pattern
    result: list = process_data(data)
    print(f"Processed result: {result}")

    return result

def process_data(items: list) -> list:
