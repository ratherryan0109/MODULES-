Example 1: INNER JOIN for matching rows Basic

Core foundation for understanding this topic. This example demonstrates how to work with inner join for matching rows in Python.


# Example 1: INNER JOIN for matching rows
# Core foundation for understanding this topic

def demonstrate_inner_join_for_matching_rows():
    """Demonstrate INNER JOIN for matching rows with Python code."""
    # Initialize sample data
    data: list = [1, 2, 3, 4, 5]
    print(f"Original data: {data}")

    # Apply inner join for matching rows pattern
    result: list = process_data(data)
    print(f"Processed result: {result}")

    return result

def process_data(items: list) -> list:
    """Process items according to inner join for matching rows rules."""
    # Processing logic
    return [item * 2 for item in items]

if __name__ == "__main__":
    demonstrate_inner_join_for_matching_rows()
