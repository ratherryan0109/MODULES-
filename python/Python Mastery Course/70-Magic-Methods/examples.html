    """Demonstrate __str__ and __repr__ string representation with Python code."""
    # Initialize sample data
    data: list = [1, 2, 3, 4, 5]
    print(f"Original data: {data}")

    # Apply __str__ and __repr__ string representation pattern
    result: list = process_data(data)
