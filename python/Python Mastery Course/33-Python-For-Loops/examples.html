# Core foundation for understanding this topic

def demonstrate_for_loop_syntax_for_x_in_iterable():
    """Demonstrate for loop syntax (for x in iterable) with Python code."""
    # Initialize sample data
    data: list = [1, 2, 3, 4, 5]
    print(f"Original data: {data}")

    # Apply for loop syntax (for x in iterable) pattern
    result: list = process_data(data)
    print(f"Processed result: {result}")

    return result

def process_data(items: list) -> list:
    """Process items according to for loop syntax (for x in iterable) rules."""
