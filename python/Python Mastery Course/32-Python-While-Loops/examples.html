Example 1: while loop syntax Basic

Core foundation for understanding this topic. This example demonstrates how to work with while loop syntax in Python.


# Example 1: while loop syntax
# Core foundation for understanding this topic

def demonstrate_while_loop_syntax():
    """Demonstrate while loop syntax with Python code."""
    # Initialize sample data
    data: list = [1, 2, 3, 4, 5]
    print(f"Original data: {data}")

    # Apply while loop syntax pattern
    result: list = process_data(data)
    print(f"Processed result: {result}")

    return result

def process_data(items: list) -> list:
    """Process items according to while loop syntax rules."""
    # Processing logic
    return [item * 2 for item in items]

if __name__ == "__main__":
    demonstrate_while_loop_syntax()
