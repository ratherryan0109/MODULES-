Example 1: Variable assignment and rebinding Basic

Core foundation for understanding this topic. This example demonstrates how to work with variable assignment and rebinding in Python.


# Example 1: Variable assignment and rebinding
# Core foundation for understanding this topic

def demonstrate_variable_assignment_and_rebinding():
    """Demonstrate Variable assignment and rebinding with Python code."""
    # Initialize sample data
    data: list = [1, 2, 3, 4, 5]
    print(f"Original data: {data}")

    # Apply variable assignment and rebinding pattern
    result: list = process_data(data)
    print(f"Processed result: {result}")

    return result

def process_data(items: list) -> list:
    """Process items according to variable assignment and rebinding rules."""
    # Processing logic
    return [item * 2 for item in items]

if __name__ == "__main__":
    demonstrate_variable_assignment_and_rebinding()
$ python example.py
