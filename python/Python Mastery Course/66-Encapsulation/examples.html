Example 1: Single underscore convention (_) Basic

Core foundation for understanding this topic. This example demonstrates how to work with single underscore convention (_) in Python.


# Example 1: Single underscore convention (_)
# Core foundation for understanding this topic

def demonstrate_single_underscore_convention():
    """Demonstrate Single underscore convention (_) with Python code."""
    # Initialize sample data
    data: list = [1, 2, 3, 4, 5]
    print(f"Original data: {data}")

    # Apply single underscore convention (_) pattern
    result: list = process_data(data)
    print(f"Processed result: {result}")

    return result

def process_data(items: list) -> list:
    """Process items according to single underscore convention (_) rules."""
    # Processing logic
    return [item * 2 for item in items]

if __name__ == "__main__":
    demonstrate_single_underscore_convention()
