# Example 1: HTTP requests for web pages
# Core foundation for understanding this topic

def demonstrate_http_requests_for_web_pages():
    """Demonstrate HTTP requests for web pages with Python code."""
    # Initialize sample data
    data: list = [1, 2, 3, 4, 5]
    print(f"Original data: {data}")

    # Apply http requests for web pages pattern
    result: list = process_data(data)
    print(f"Processed result: {result}")

    return result
