# Example 1: Opening text files (mode="r", "w", "a")
# Core foundation for understanding this topic

def demonstrate_opening_text_files_mode_r_w_a():
    """Demonstrate Opening text files (mode="r", "w", "a") with Python code."""
    # Initialize sample data
    data: list = [1, 2, 3, 4, 5]
    print(f"Original data: {data}")

    # Apply opening text files (mode="r", "w", "a") pattern
    result: list = process_data(data)
    print(f"Processed result: {result}")

    return result

    """Process items according to opening text files (mode="r", "w", "a") rules."""
