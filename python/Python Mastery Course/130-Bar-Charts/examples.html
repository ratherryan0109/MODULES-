    """Demonstrate plt.bar() vertical bars with Python code."""
    # Initialize sample data
    data: list = [1, 2, 3, 4, 5]
    print(f"Original data: {data}")

    # Apply plt.bar() vertical bars pattern
    result: list = process_data(data)
