# Example 1: plt.plot() for line charts

def demonstrate_plt_plot_for_line_charts():
    """Demonstrate plt.plot() for line charts with Python code."""
    # Initialize sample data
    data: list = [1, 2, 3, 4, 5]
    print(f"Original data: {data}")

    # Apply plt.plot() for line charts pattern
    result: list = process_data(data)

    return result
