1. Basic Shapes

ctx.fillRect(50, 50, 100, 80);
ctx.strokeRect(200, 50, 100, 80);
ctx.clearRect(70, 70, 60, 40);

// Triangle
ctx.beginPath();
ctx.moveTo(50, 200);
ctx.lineTo(150, 250);
ctx.lineTo(20, 250);
ctx.closePath();
ctx.fill();

2. Gradients

const linear = ctx.createLinearGradient(0, 0, 400, 0);
linear.addColorStop(0, 'red');
linear.addColorStop(0.5, 'yellow');
linear.addColorStop(1, 'blue');
ctx.fillStyle = linear;
ctx.fillRect(0, 0, 400, 150);

const radial = ctx.createRadialGradient(200, 225, 10, 200, 225, 80);
radial.addColorStop(0, 'white');
radial.addColorStop(1, 'purple');
ctx.fillStyle = radial;
ctx.beginPath();
ctx.arc(200, 225, 80, 0, Math.PI * 2);
ctx.fill();

3. Text Rendering

ctx.font = 'bold 36px Arial';
ctx.fillStyle = '#e74c3c';
ctx.textAlign = 'center';
ctx.textBaseline = 'middle';
ctx.fillText('Hello Canvas!', 200, 60);

ctx.font = '24px Courier New';
ctx.strokeStyle = '#2c3e50';
ctx.lineWidth = 1;
ctx.strokeText('Stroked Text', 200, 130);

ctx.font = '16px Georgia';
ctx.fillStyle = '#27ae60';
ctx.fillText('Rendered at Retina quality', 200, 175);

4. Path Operations

ctx.beginPath();
ctx.arc(100, 100, 60, 0, Math.PI * 2);
ctx.fillStyle = '#3498db';
ctx.fill();
ctx.strokeStyle = '#2c3e50';
ctx.lineWidth = 3;
ctx.stroke();

// Bezier curve
ctx.beginPath();
ctx.moveTo(200, 200);
ctx.bezierCurveTo(300, 50, 350, 250, 380, 100);
ctx.strokeStyle = '#e74c3c';
ctx.lineWidth = 4;
ctx.stroke();

6. Animation

function animate() {
  ctx.clearRect(0, 0, 400, 300);
  // Draw moving circle
  ctx.beginPath();
  ctx.arc(x, 150, 30, 0, Math.PI * 2);
  ctx.fillStyle = '#e74c3c';
  ctx.fill();
  x += 2;
  if (x > 400) x = 0;
  animId = requestAnimationFrame(animate);
}
