4. Media Query Breakpoints

/* Mobile-first: base mobile styles */
body { font-size: 16px; }

/* Tablet >= 768px */
@media (min-width: 768px) {
  body { font-size: 18px; }
}

/* Desktop >= 1024px */
@media (min-width: 1024px) {
  body { font-size: 20px; }
}

/* Orientation */
@media (orientation: landscape) { }
@media (prefers-color-scheme: dark) { }
@media (prefers-reduced-motion: reduce) { }
